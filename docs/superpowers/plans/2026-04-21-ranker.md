# ranker.py Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `ranker.py` — a pure-Python TF-IDF re-ranker that re-orders `decode_topn` candidates by max cosine similarity to individual `endfield_dict.txt` entries.

**Architecture:** `ranker.py` is a standalone module (no external deps) exposing three functions: `clean_punctuation`, `build_ranker`, and `rerank`. `run.py` calls `build_ranker` once at startup and `rerank` after each `decode_topn` call. `decoder.py` imports `clean_punctuation` for display only.

**Tech Stack:** Python stdlib only — `math`, `collections`, `re`.

```python
import re
import math
from collections import Counter

_KEEP = {'——', '……'}

def clean_punctuation(text: str) -> str:
    # Temporarily replace preserved sequences
    text = text.replace('——', '\x00').replace('……', '\x01')
    # Strip Chinese and ASCII punctuation
    text = re.sub(r'[，。！？、；：""''（）【】《》「」『』…—~!?,.:;\'\"()\[\]{}<>]', '', text)
    # Restore preserved sequences
    return text.replace('\x00', '——').replace('\x01', '……')
```

```python
def _ngrams(text, ns=(1, 2)):
    grams = []
    for n in ns:
        for i in range(len(text) - n + 1):
            grams.append(text[i:i+n])
    return grams

def _tf_vector(grams):
    counts = Counter(grams)
    total = len(grams)
    return {g: c / total for g, c in counts.items()} if total else {}

class Ranker:
    def __init__(self, idf, entries):
        self.idf = idf          # {ngram: float}
        self.entries = entries  # [(weight_vec, l2_norm), ...]

def build_ranker(dict_path: str) -> Ranker:
    try:
        with open(dict_path, encoding='utf-8') as f:
            lines = [l.strip() for l in f if l.strip()]
    except FileNotFoundError:
        raise FileNotFoundError(f'Dictionary not found: {dict_path}')
    if not lines:
        raise FileNotFoundError(f'Dictionary is empty: {dict_path}')

    N = len(lines)
    df: Counter = Counter()
    entry_grams = []
    for line in lines:
        grams = _ngrams(line)
        entry_grams.append(grams)
        for g in set(grams):
            df[g] += 1

    idf = {g: math.log((N + 1) / (cnt + 1)) for g, cnt in df.items()}

    entries = []
    for grams in entry_grams:
        tf = _tf_vector(grams)
        vec = {g: tf[g] * idf.get(g, 0.0) for g in tf}
        norm = math.sqrt(sum(v * v for v in vec.values())) or 0.0
        entries.append((vec, norm))

    return Ranker(idf, entries)
```

```python
def _candidate_string(segments):
    parts = []
    for _key, words in segments:
        parts.append(words[0] if words else '')
    return ''.join(parts)

def _cosine(cand_vec, cand_norm, entry_vec, entry_norm):
    if cand_norm == 0.0 or entry_norm == 0.0:
        return 0.0
    dot = sum(cand_vec.get(g, 0.0) * w for g, w in entry_vec.items())
    return dot / (cand_norm * entry_norm)

def rerank(candidates, ranker: Ranker):
    if len(candidates) <= 1:
        return candidates

    def tfidf_score(candidate):
        dp_score, segments = candidate
        text = clean_punctuation(_candidate_string(segments))
        grams = _ngrams(text)
        tf = _tf_vector(grams)
        vec = {g: tf[g] * ranker.idf.get(g, 0.0) for g in tf}
        norm = math.sqrt(sum(v * v for v in vec.values())) or 0.0
        best = max(
            (_cosine(vec, norm, ev, en) for ev, en in ranker.entries),
            default=0.0
        )
        return (best, dp_score)

    return sorted(candidates, key=tfidf_score, reverse=True)
```

```python
        s = input('请输入密文字母串: ').strip()

            print()

            continue
        segs = decode(s.lower(), d)
        if segs is None:
            print('  [!] 无法完整解码，逐字母查找:')
            for ch in s.lower():
                entry = d.get(ch, {})

        else:
            best = []

                top = words[0]

            if _HAS_TOPN:
```
